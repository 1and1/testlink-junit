16  package net.oneandone.testlinkjunit.tljunit;
17  
18  import java.lang.annotation.ElementType;
19  import java.lang.annotation.Retention;
20  import java.lang.annotation.RetentionPolicy;
21  import java.lang.annotation.Target;
22  
23  /**
24   * Annotation for marking {@link org.junit.Test}s the results of which should be reported in testlink xml file.
25   * 
26   * <ul>
27   * <li>
28   * Anotate your tests with {@link TestLink} and provide either an {@link TestLink#externalId()} or
29   * {@link TestLink#internalId()}, one of which is required.</li>
30   * <li>
31   * Tests annotated with {@link org.junit.Ignore} as well as tests with failing assumptions will be reported as
32   * <tt>BLOCKED</tt>, see {@link org.junit.Assume} as well.</li>
33   * <li>Failing tests or tests in error are reported as <tt>FAILED</tt>.</li>
34   * </ul>
35   * 
36   * <pre>
37   * &#064;Test
38   * &#064;TestLink(internalId = 1)
39   * public void testPassed() {
40   *     assertTrue(true); // PASSED
41   * }
42   * </pre>
43   * 
44   * <pre>
45   * &#064;Test
46   * &#064;TestLink(externalId = &quot;PROJECT-1&quot;)
47   * public void testExternalId() {
48   *     assertTrue(false); // FAILED
49   * }
50   * </pre>
51   *
52   * <pre>
53   * &#064;Test
54   * &#064;TestLink(internalId = 2)
55   * &#064;Ignore(&quot;Just ignore this&quot;)
56   * public void testIgnored() {
57   *     assertTrue(true); // BLOCKED as test is ignored.
58   * }
59   * </pre>
60   * 
61   * <pre>
62   * &#064;Test
63   * &#064;TestLink(externalId = &quot;ASSUMPTION_FAILED&quot;)
64   * public void testWithFailingAssumption() {
65   *     assumeTrue(false); // BLOCKED as assumption failed.
66   * }
67   * </pre>
68   * 
69   * @author Mirko Friedenhagen
70   */
71  @Retention(RetentionPolicy.RUNTIME)
72  @Target(ElementType.METHOD)
73  public @interface TestLink {
74  
75      /** Default value for missing externalIds. */
76      String NOT_AVAILABLE = "NOT_AVAILABLE";
77  
78      /**
79       * internal_id of a Test to be reported.
80       */
81      long internalId() default 0;
82  
83      /**
84       * external_id of a Test to be reported.
85       */
86      String externalId() default NOT_AVAILABLE;
87  }
